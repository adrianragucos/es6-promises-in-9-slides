const delay = ((ms) => {
    return new Promise((resolve, reject) => {
        if (typeof ms === 'number') {
            setTimeout(resolve, ms);
        } else {
            reject();
        }
    });
});

const success = (() => {
    console.log('3 seconds');
});

const redirect = (() => {
    setTimeout(() => {
        window.location.replace('http://bit.ly/2jbiAYH');
    }, 1000);    
});

const failure = (() => {
    alert('not a number!!!!11!11!one');
});

delay(1000).then(success).then(redirect).catch(failure);
