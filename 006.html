promises

we can also use .catch if we want to be idiomatic:

const delay = ((ms) => {
    return new Promise((resolve, reject) => {
        if (typeof ms === 'number') {
            setTimeout(resolve, ms);
        } else {
            reject();
        }
    });
});

const success = (() => {
    console.log('3 seconds');
});

const failure = (() => {
    alert('not a number!!!!11!11!one');
});

delay('foo').then(success).catch(failure);

.catch is effectively .then(null, failure)
