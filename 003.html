promises

example:

const delay = ((ms) => {
    return new Promise((resolve, reject) => {
        if (typeof ms === 'number') {
            setTimeout(resolve, ms);
        } else {
            reject();
        }        
    });
});

const success = (() => {
    console.log('3 seconds');
});

delay(3000).then(success);
